CREATE TABLE lecture_table (
    lec_no           NUMBER(38) PRIMARY KEY,
    lec_parentno     NUMBER(38) DEFAULT 0,
    lec_title        VARCHAR2(50),
    lec_price        NUMBER(38),
    lec_content      VARCHAR2(4000),
    lec_imgfile      VARCHAR2(50),
    lec_spofile      VARCHAR2(50),
    lec_uploaddate   DATE DEFAULT SYSDATE
);

SELECT
    *
FROM
    (
        SELECT
            ROWNUM AS recnum,
            lvl,
            lec_no,
            lec_parentno,
            lec_title,
            lec_price,
            lec_imgfile,
            lec_spofile
        FROM
            (
                SELECT
                    level AS lvl,
                    lec_no,
                    lec_parentno,
                    lec_title,
                    lec_price,
                    lec_imgfile,
                    lec_spofile
                FROM
                    lecture_table
                START WITH
                    lec_parentno = 0
                CONNECT BY
                    PRIOR lec_no = lec_parentno
                ORDER SIBLINGS BY
                    lec_no DESC
            )
    )
WHERE
    recnum BETWEEN ( 1 - 1 ) * 100 + ( 1 - 1 ) * 10 + 1 AND ( 1 - 1 ) * 100 + 1 * 10;

SELECT
    *
FROM
    lecture_table;
